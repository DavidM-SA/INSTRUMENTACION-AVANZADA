143void signalCode(void *pvParameters)
144{
145 for (;;)
146 {
147 // leemos datos conversor analogico
148 uint32_t reading = adc1_get_raw(ADC1_CHANNEL_7);
149 int voltaje = esp_adc_cal_raw_to_voltage(4095 - reading, &adc_chars);
150
151 // mandamos dato a la cola
152 if (xQueueGenericSend(colaSignal, (void *)&voltaje, (TickType_t)10, queueSEND_TO_BACK) != pdPASS)
153 {
154 Serial.println("No se ha podido enviar el dato sin filtrar a la cola");
155 }
156 //asegurarse que se lean los datos a 1Mhz
157 vTaskDelay(1);
158 }
159}

166void filterCode(void *pvParameters)
167{
168 for (;;)
169 {
170 int signalR;
171 if (xQueueReceive(colaSignal, &(signalR), (TickType_t)8) != pdTRUE)
172 {
173 Serial.println("No se ha podido recibir el dato de la cola");
174 }
175
176 SampleFilter_put(&filterFIR, signalR);
177 float filtro = SampleFilter_get(&filterFIR)/1000;
178
179 if (xQueueGenericSend(colaFiltro, (void *)&filtro, (TickType_t)10, queueSEND_TO_BACK) != pdPASS)
180 {
181 Serial.println("No se ha podido enviar el dato filtrado a la cola");
182 }
183 }
184}

191void Signal_Values_Code(void *pvParameters)
192{
193 for (;;)
194 {
195 float value;
196 if (xQueueReceive(colaFiltro, &(value), (TickType_t)8) != pdTRUE)
197 {
198 Serial.println("No se ha podido recibir para media");
199 }
200 else
201 {
202 // Sacamos la media
203 averaje = averaje + value;
204
205 // sacamos el maximo
206 if (value > maxim)
207 {
208 maxim = value;
209 }
210
211 // sacamos el minimo
212 if (value < minimim)
213 {
214 minimim = value;
215 }
216 repet++;
217 if (repet > 1000)
218 {
219 if (client.isMqttConnected())
220 {
221 client.publish("Eupla806131/filtros/media", String(averaje / (1000)));
222 client.publish("Eupla806131/filtros/max", String(maxim));
223 client.publish("Eupla806131/filtros/min", String(minimim));
224 }
225 repet = 0;
226 averaje = 0;
227 maxim = 0;
228 minimim = 4500;
229 }
230 }
